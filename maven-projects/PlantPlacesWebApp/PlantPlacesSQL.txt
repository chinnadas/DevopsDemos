CREATE DATABASE PlantPlaces;

USE PlantPlaces;

CREATE TABLE IF NOT EXISTS `plants` (
  `PLANT_ID` int(11) NOT NULL AUTO_INCREMENT,
  `GENUS` varchar(255) DEFAULT NULL,
  `SPECIES` varchar(255) DEFAULT NULL,
  `CULTIVAR` varchar(255) DEFAULT NULL,
  `COMMON` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`PLANT_ID`)
) ENGINE=InnoDB  DEFAULT CHARSET=latin1 AUTO_INCREMENT=8 ;

CREATE TABLE IF NOT EXISTS `specimens` (
  `SPECIMEN_ID` int(11) NOT NULL AUTO_INCREMENT,
  `PLANT_ID` int(11) NOT NULL,
  `LATITUDE` varchar(60) DEFAULT NULL,
  `LONGITUDE` varchar(60) DEFAULT NULL,
  `PLANTED_BY` varchar(255) DEFAULT NULL,
  `PLANTED_DATE` varchar(255) DEFAULT NULL,
  `DESCRIPTION` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`SPECIMEN_ID`)
) ENGINE=InnoDB  DEFAULT CHARSET=latin1 AUTO_INCREMENT=5 ;	

CREATE TABLE IF NOT EXISTS `photos` (
  `PHOTO_ID` int(11) NOT NULL AUTO_INCREMENT,
  `SPECIMEN_ID` int(11) NOT NULL,
  `URI` varchar(511) DEFAULT NULL,
  `DATE_TAKEN` varchar(60) DEFAULT NULL,
  `CONTRIBUTOR` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`PHOTO_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1 AUTO_INCREMENT=1 ;
